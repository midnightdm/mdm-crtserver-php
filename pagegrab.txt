					return false;

					setCookie("usr_lang",$(this).data("lang"),365);
					return true;
				}
				if ( $( this ).hasClass( "chosen-default" ) )
					return false;
				if ( $( this ).hasClass( "chosen-single" ) )
					return false;
				
				if ( $( this ).hasClass( "close_login" ) )
				{
					close_login_window();
				}
				
				if ( $( this ).hasClass( "ajax" ) )
				{
					create_alert( $(this).attr('href') );
					return false;
				}
				
				if ( $( this ).hasClass( "ajax-black-refresh" ) )
				{
					create_alert( $(this).attr('href') , true, null,true);
					return false;
				}
				
				if ( $( this ).hasClass( "ajax-black" ) )
				{
					create_alert( $(this).attr('href') , true);
					return false;
				}
				
				if ( $( this ).hasClass( "preview-map" ) )
				{
					preview_map_init(this);
					return false;
				}
				
				if(this.href=="")
					return true;
				//if(this.href=="")
				//	return false;
				
				var a = new RegExp('/' + window.location.host + '/');
				if(!a.test(this.href)) {
					event.preventDefault();
					event.stopPropagation();
					window.open(this.href, '_blank');
					return false;
				}
				
				event.preventDefault(); 
				
				var url = $(this).attr('href');
				if(url=="#")
					return true;
					
				
				removeAds();
				observer.disconnect();
				var scroll_pos = $(window).scrollTop();
				$('#content_in').html("");
				
				
				
				
				if(history.pushState) {
					var stateData = history.state;
					if(stateData==null){
						stateData = {
							path: window.location.href,
							scrollTop: scroll_pos
						};
					}
					stateData.scrollTop = scroll_pos;
					window.history.replaceState(stateData, 'MyShipTracking', stateData.path);
				
					var stateData = {
						path: url,
						scrollTop: 0
					};
					history.pushState(stateData, null, url);
				}
				load_page(url);
			});
